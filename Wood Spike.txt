<script>
const HOURS = [10,11,12,13,14,15,16,17];
const DAYS = ['Sun','Mon','Tue','Wed','Thu','Fri','Sat'];
const COLORS = ['#B5D4F4','#9FE1CB','#FAC775','#F4C0D1'];
const TEXT_COLORS = ['#0C447C','#085041','#633806','#72243E'];

let weekOffset = 0;
let currentPlayerIdx = 0;
let dragState = null;

let players = [
  {name:'Player 1', color:COLORS[0], textColor:TEXT_COLORS[0], selections:{}, busy:{}},
  {name:'Player 2', color:COLORS[1], textColor:TEXT_COLORS[1], selections:{}, busy:{}},
  {name:'Player 3', color:COLORS[2], textColor:TEXT_COLORS[2], selections:{}, busy:{}},
  {name:'Player 4', color:COLORS[3], textColor:TEXT_COLORS[3], selections:{}, busy:{}}
];

function getWeekStart(){
  const d=new Date(); d.setHours(0,0,0,0);
  d.setDate(d.getDate()-d.getDay()+weekOffset*7);
  return d;
}

function slotKey(d,h){ return d+'_'+h; }

function isPlayerBusy(pi,d,h){
  const ranges=(players[pi].busy[d])||[];
  return ranges.some(([s,e])=>h>=s&&h<e);
}

function countPlayers(d,h){
  const key=slotKey(d,h);
  return players.filter(p=>p.selections[key]).length;
}

function isLocked(d,h){ return countPlayers(d,h)>=4; }

function canToggle(pi,d,h){
  if(isPlayerBusy(pi,d,h)) return false;
  const key=slotKey(d,h);
  if(isLocked(d,h)&&!players[pi].selections[key]) return false;
  return true;
}

function setSlot(pi,d,h,val){
  if(!canToggle(pi,d,h)) return;
  const key=slotKey(d,h);
  if(val) players[pi].selections[key]=true;
  else delete players[pi].selections[key];
}

function render(){
  const ws=getWeekStart();
  const we=new Date(ws); we.setDate(we.getDate()+6);
  document.getElementById('week-label').textContent=
    ws.toLocaleDateString('en-US',{month:'short',day:'numeric'})+' – '+
    we.toLocaleDateString('en-US',{month:'short',day:'numeric',year:'numeric'});

  const today=new Date(); today.setHours(0,0,0,0);
  const g=document.getElementById('grid'); g.innerHTML='';

  const corner=document.createElement('div'); corner.className='corner'; g.appendChild(corner);

  for(let d=0;d<7;d++){
    const dt=new Date(ws); dt.setDate(dt.getDate()+d);
    const isToday=dt.getTime()===today.getTime();
    const hdr=document.createElement('div');
    hdr.className='day-hdr'+(isToday?' today':'');
    hdr.innerHTML=`<div class="dow">${DAYS[d]}</div><div class="date">${dt.getDate()}</div>`;
    g.appendChild(hdr);
  }

  HOURS.forEach(hour=>{
    const tc=document.createElement('div'); tc.className='time-cell';
    tc.textContent=hour===12?'12p':hour<12?hour+'a':(hour-12)+'p';
    g.appendChild(tc);

    for(let d=0;d<7;d++){
      const key=slotKey(d,hour);
      const busy=isPlayerBusy(currentPlayerIdx,d,hour);
      const locked=isLocked(d,hour);
      const selected=!!players[currentPlayerIdx].selections[key];
      const cnt=countPlayers(d,hour);

      const slot=document.createElement('div');
      slot.className='slot '+(locked&&!selected?'locked':busy?'busy':selected?'selected':'free');
      slot.dataset.day=d; slot.dataset.hour=hour;

      if(busy){
        const lbl=document.createElement('div'); lbl.className='slot-label'; lbl.textContent='busy'; slot.appendChild(lbl);
      } else if(locked&&!selected){
        const lbl=document.createElement('div'); lbl.className='slot-label';
        lbl.innerHTML='<i class="ti ti-bolt" aria-hidden="true"></i>'; slot.appendChild(lbl);
      } else if(selected){
        const ic=document.createElement('div'); ic.className='slot-label';
        ic.innerHTML='<i class="ti ti-check" aria-hidden="true"></i>'; slot.appendChild(ic);
      }

      if(cnt>0&&!locked&&!selected){
        const badge=document.createElement('div'); badge.className='cnt-badge';
        badge.style.color=cnt>=3?'#D85A30':'var(--color-text-tertiary)';
        badge.textContent=cnt+'✓'; slot.appendChild(badge);
      }

      g.appendChild(slot);
    }
  });

  attachDragListeners();
  renderPlayers();
  renderMatches();
}

function attachDragListeners(){
  const g=document.getElementById('grid');

  function slotFromEl(el){
    const s=el&&el.closest&&el.closest('.slot');
    if(!s) return null;
    return {el:s,day:+s.dataset.day,hour:+s.dataset.hour};
  }

  function startDrag(e,slot){
    if(!slot) return;
    if(isPlayerBusy(currentPlayerIdx,slot.day,slot.hour)) return;
    const key=slotKey(slot.day,slot.hour);
    const wasSelected=!!players[currentPlayerIdx].selections[key];
    dragState={painting:!wasSelected};
    setSlot(currentPlayerIdx,slot.day,slot.hour,dragState.painting);
    render();
  }

  function duringDrag(e){
    if(!dragState) return;
    let el;
    if(e.touches){ const t=e.touches[0]; el=document.elementFromPoint(t.clientX,t.clientY); e.preventDefault(); }
    else { el=e.target; }
    const slot=slotFromEl(el); if(!slot) return;
    const key=slotKey(slot.day,slot.hour);
    const cur=!!players[currentPlayerIdx].selections[key];
    if(cur===dragState.painting) return;
    setSlot(currentPlayerIdx,slot.day,slot.hour,dragState.painting);
    render();
  }

  function endDrag(){ dragState=null; }

  g.addEventListener('mousedown',e=>{ if(e.button!==0) return; startDrag(e,slotFromEl(e.target)); });
  g.addEventListener('mouseover',duringDrag);
  g.addEventListener('mouseup',endDrag);
  g.addEventListener('mouseleave',endDrag);
  g.addEventListener('touchstart',e=>{ const t=e.touches[0]; startDrag(e,slotFromEl(document.elementFromPoint(t.clientX,t.clientY))); },{passive:false});
  g.addEventListener('touchmove',duringDrag,{passive:false});
  g.addEventListener('touchend',endDrag);
}

function renderPlayers(){
  const pg=document.getElementById('player-grid'); pg.innerHTML='';
  players.forEach((p,i)=>{
    const cnt=Object.keys(p.selections).length;
    const card=document.createElement('div');
    card.className='player-card'+(i===currentPlayerIdx?' me':'');
    card.innerHTML=`<div class="avatar" style="background:${p.color};color:${p.textColor}">${p.name.charAt(0)}</div>
      <div class="player-info"><div class="name">${p.name}</div><div class="slots-count">${cnt} slot${cnt!==1?'s':''} picked</div></div>`;
    pg.appendChild(card);
  });
}

function renderMatches(){
  const ws=getWeekStart();
  const m=document.getElementById('matches');
  const all=[];
  for(let d=0;d<7;d++) HOURS.forEach(h=>{
    const cnt=countPlayers(d,h);
    if(cnt>=2){
      const dt=new Date(ws); dt.setDate(dt.getDate()+d);
      const label=dt.toLocaleDateString('en-US',{weekday:'short',month:'short',day:'numeric'});
      const timeStr=h===12?'12:00 pm':h<12?`${h}:00 am`:`${h-12}:00 pm`;
      const who=players.filter(p=>p.selections[slotKey(d,h)]).map(p=>p.name).join(', ');
      all.push({label,timeStr,cnt,who,sort:cnt>=4?0:cnt===3?1:2});
    }
  });
  all.sort((a,b)=>a.sort-b.sort);
  if(!all.length){ m.innerHTML='<div class="empty-state">No overlaps yet — get your crew to pick their slots!</div>'; return; }
  m.innerHTML=all.slice(0,6).map(e=>`<div class="match-slot">
    <div class="match-info"><div class="time">${e.label} · ${e.timeStr}</div><div class="who">${e.who}</div></div>
    <span class="match-pill ${e.cnt>=4?'pill-4':e.cnt===3?'pill-3':'pill-low'}">${e.cnt}/4</span>
  </div>`).join('');
}

function switchPlayer(idx){
  currentPlayerIdx=idx;
  for(let i=0;i<4;i++) document.getElementById('btn-'+i).classList.toggle('active',i===idx);
  render();
}

function shiftWeek(n){ weekOffset+=n; render(); }

function openModal(){ document.getElementById('modal').classList.add('open'); document.getElementById('new-name').focus(); }
function closeModal(){ document.getElementById('modal').classList.remove('open'); document.getElementById('new-name').value=''; }
function confirmAdd(){
  const name=document.getElementById('new-name').value.trim(); if(!name) return;
  players[currentPlayerIdx].name=name;
  document.getElementById('btn-'+currentPlayerIdx).textContent=name;
  closeModal(); render();
}

function copyNag(){
  const ws=getWeekStart();
  const slots3=[];
  for(let d=0;d<7;d++) HOURS.forEach(h=>{ if(countPlayers(d,h)===3){ const dt=new Date(ws); dt.setDate(dt.getDate()+d); slots3.push(dt.toLocaleDateString('en-US',{weekday:'short',month:'short',day:'numeric'})+' at '+(h===12?'12:00 pm':h<12?`${h}:00 am`:`${h-12}:00 pm`)); } });
  const msg=slots3.length>0
    ?`YO we have 3/4 confirmed for SpikeBall on Wood Spike — we just need ONE more to lock it in! 🏐\n\nTimes that work for 3 of us:\n${slots3.slice(0,3).map(s=>'• '+s).join('\n')}\n\nPick your slots or you're carrying the net: [your Wood Spike link]`
    :`SPIKEBALL SCHEDULING IS HAPPENING AND YOU NEED TO PARTICIPATE 🏐\n\nI built an entire app called Wood Spike because our group chat is a disaster.\n\nTakes 30 seconds. Click your free times. Do it:\n[your Wood Spike link]\n\nYou have 24 hours before I make unilateral decisions.`;
  navigator.clipboard.writeText(msg).then(()=>{
    const btn=document.getElementById('nag-btn');
    btn.textContent='✓ Copied!'; btn.classList.add('copied');
    setTimeout(()=>{ btn.innerHTML='<i class="ti ti-send" aria-hidden="true"></i> Copy nag message'; btn.classList.remove('copied'); },2000);
  }).catch(()=>alert(msg));
}

render();
document.addEventListener('mouseup',()=>{ dragState=null; });
document.getElementById('new-name').addEventListener('keydown',e=>{ if(e.key==='Enter') confirmAdd(); if(e.key==='Escape') closeModal(); });
</script>
